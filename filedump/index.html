this uses sdfs. which stands for something(?) distance fields. idk the s. idk what distance fields are! when i google it all the results are about 3d!
but i got it working anyways. heres the script for the godot material, thats the main point of all this:
shader_type spatial;

uniform sampler2D outline;
uniform sampler2D fill;
uniform bool nofill;

void fragment() {
	vec4 soutline = texture(outline,UV);
	soutline = smoothstep(.499,.5,soutline);
	vec4 final = mix(texture(fill,UV),vec4(0,0,0,1),soutline.r);
	if (nofill) {final = mix(vec4(0,0,0,0),vec4(0,0,0,1),soutline.r);}
	ALBEDO = final.xyz;
	ALPHA = final.a;
}
